CREATE SEQUENCE SEQ_FACULTATE_ID
START WITH 1
INCREMENT BY 1
MAXVALUE 99999
NOCYCLE
NOCACHE;

CREATE TABLE FACULTATE
(
 ID number (5,0)
 CONSTRAINT FACULTATE_PK PRIMARY KEY,
 NUME VARCHAR2 (50)
 CONSTRAINT FACULTATE_NUME_NN NOT NULL
 CONSTRAINT FACULTATE_NUME_U UNIQUE,
 ADRESA VARCHAR2 (100)
 CONSTRAINT FACULTATE_ADRESA_NN NOT NULL,
 TELEFON VARCHAR2 (15)
 CONSTRAINT FACULTATE_TELEFON_NN NOT NULL
 CONSTRAINT FACULTATE_TELEFON_U UNIQUE,
 EMAIL VARCHAR2 (50)
 CONSTRAINT FACULTATE_EMAIL_NN NOT NULL
 CONSTRAINT FACULTATE_EMAIL_U UNIQUE
);

--CREAREA TABELULUI PROGRAMA

CREATE TABLE PROGRAMA
(
 SERIE_ID number (5,0)
 CONSTRAINT PROGRAMA_SERIE_FK REFERENCES SERIE (ID) ON DELETE CASCADE,
 DISCIPLINA_ID number (5,0)
 CONSTRAINT PROGRAMA_DISCIPLINA_FK REFERENCES DISCIPLINA (ID) ON DELETE CASCADE,
 NR_ORE NUMBER(3,0)
 CONSTRAINT PROGRAMA_ORE_NN NOT NULL,
 NR_CREDITE NUMBER(3,0)
 CONSTRAINT PROGRAMA_CREDITE_NN NOT NULL,
 CONSTRAINT PROGRAMA_PK PRIMARY KEY (SERIE_ID, DISCIPLINA_ID)
);


--CREAREA TABELULUI CONTRACT_DE_STUDII

CREATE TABLE CONTRACT_DE_STUDII
(
 GRUPA_ID number (5,0)
 CONSTRAINT CONTRACT_DE_STUDII_GRUPA_FK REFERENCES GRUPA (ID) ON DELETE CASCADE,
 STUDENT_ID number (5,0)
 CONSTRAINT CONTRACT_DE_STUDII_STUDENT_FK REFERENCES STUDENT (ID) ON DELETE CASCADE,
 FORMA_FINANTARE VARCHAR2 (10)
 CONSTRAINT CONTRACT_DE_STUDII_FORMAFINANTARE_NN NOT NULL
 CONSTRAINT CONTRACT_DE_STUDII_FORMAFINANTARE_CK CHECK (FORMA_FINANTARE IN ('buget', 'taxa')),
 DATA_INCEPERE DATE,
 DATA_FINALIZARE DATE,
 CONSTRAINT CONTRACT_DE_STUDII_PK PRIMARY KEY (GRUPA_ID, STUDENT_ID),
 CONSTRAINT CONTRACT_DE_STUDII_DATE_CK CHECK (DATA_INCEPERE < DATA_FINALIZARE)
);

--CREAREA TABELULUI PREDARE

CREATE TABLE PREDARE
(
 PROFESOR_ID number (5,0)
 CONSTRAINT PREDARE_PROFESOR_FK REFERENCES PROFESOR (ID) ON DELETE CASCADE,
 DISCIPLINA_ID number (5,0)
 CONSTRAINT PREDARE_DISCIPLINA_FK REFERENCES DISCIPLINA (ID) ON DELETE CASCADE,
 SALA INT,
 CONSTRAINT PREDARE_PK PRIMARY KEY (PROFESOR_ID, DISCIPLINA_ID)
);
